def go(*args)
  begin
    result = @function.call(*args)
  rescue => error
  end

  # this contorted-looking code is to avoid swallowing exceptions that
  # are raised by #succeed itself (e.g. if someone passes a buggy
  # callback that raises an exception).
  if error
    fail(error)
  else
    succeed(result)
  end
end
